// Додаємо обробник для кнопки Close document.querySelector('#editRolesModal .btn-close, #editRolesModal .btn-secondary').addEventListener('click', function() { const systemId = document.getElementById('systemSelect').value; if (systemId) { const languagePrefix = window.location.pathname.split('/')[1]; $.ajax({ url: `/${languagePrefix}/app_access/roles/system/?system_id=${systemId}`, type: 'GET', success: function(response) { if (response.roles) { updateRolesList(response.roles); document.getElementById('defaultRolesSection').style.display = 'block'; } }, error: function(error) { console.error('Error loading roles:', error); toastr.error('{% trans "Error loading roles" %}'); } }); } });
